def zodiac1(year):
    animals = ["Monkey", "Rooster", "Dog", "Pig", "Rat", "Ox", "Tiger", "Rabbit", "Dragon", "Snake", "Horse", "Sheep"]

    if year >= 1900:
        year2 = (year - 1900) % 12
        return animals[year2]
    else:
        return "Year should be 1900 or later."

year = int(input("Enter a year: "))
zodiac2 = zodiac1(year)

if zodiac2 != "Year should be 1900 or later.":
    print(f"The Chinese zodiac sign for the year {year} is {zodiac2}.")
else:
    print(zodiac2)
